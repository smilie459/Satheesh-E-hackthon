    <script>
          // Login Function 
        function checkLogin() {
            const user = document.getElementById('userInput').value;
            const pass = document.getElementById('passInput').value;
            const error = document.getElementById('loginError');

            if (user === "Satheesh" && pass === "Satheesh@123") {
                document.getElementById('loginOverlay').style.display = 'none';
                document.getElementById('appContent').style.display = 'block';
                initCharts(); 
                // Initialize charts after login
                render();
            } else {
                error.style.display = 'block';
            }
        }

        let tasks = [
            { name: "Math Assignment", status: "Completed" },
            { name: "Science Project", status: "Pending" },
            { name: "English Essay", status: "Overdue" }
        ];

        function showPage(pId, el) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            document.getElementById(pId).classList.add('active');
            document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
            el.classList.add('active');
        }

        function render() {
            const tList = document.getElementById('taskList');
            const recentList = document.getElementById('recentActivityList');
            if(tList) tList.innerHTML = "";
            if(recentList) recentList.innerHTML = "";
            
            let counts = { Completed: 0, Pending: 0, Overdue: 0 };
            tasks.forEach((t, i) => {
                counts[t.status]++;
                const col = t.status === "Completed" ? "var(--green)" : (t.status === "Overdue" ? "var(--red)" : "var(--blue)");
                if(tList) tList.innerHTML += `<div class="task-card" style="border-left-color:${col}"><div><b>${t.name}</b><br><small>${t.status}</small></div><div><button class="btn btn-edit" onclick="editTask(${i})">Edit</button><button class="btn btn-red" onclick="deleteTask(${i})">Delete</button></div></div>`;
                if(recentList && i < 3) recentList.innerHTML += `<div style="padding:10px 0; border-bottom:1px solid #eee; display:flex; justify-content:space-between; color: #333;"><span>${t.name}</span><b style="color:${col}">${t.status}</b></div>`;
            });

            if(document.getElementById('totalCount')) {
                document.getElementById('totalCount').innerText = tasks.length;
                document.getElementById('completeCount').innerText = counts.Completed;
                document.getElementById('pendingCount').innerText = counts.Pending;
                document.getElementById('overdueCount').innerText = counts.Overdue;
                let rate = tasks.length > 0 ? Math.round((counts.Completed / tasks.length) * 100) : 0;
                document.getElementById('rateText').innerText = rate + "%";
                document.getElementById('rateBar').style.width = rate + "%";
                document.getElementById('dashTotal').innerText = tasks.length;
                document.getElementById('totalScore').innerText = Math.round((92 + rate) / 2) + "%";
            }
        }

        function openModal() { document.getElementById('editIndex').value = ""; document.getElementById('tName').value = ""; document.getElementById('modal').style.display='flex'; }
        function closeModal() { document.getElementById('modal').style.display='none'; }
        
        function saveTask() {
            const name = document.getElementById('tName').value;
            const status = document.getElementById('tStatus').value;
            const idx = document.getElementById('editIndex').value;
            if(!name) return;
            if(idx === "") tasks.unshift({ name, status });
            else { tasks[idx].name = name; tasks[idx].status = status; }
            render(); closeModal();
        }

        function editTask(i) { document.getElementById('editIndex').value = i; document.getElementById('tName').value = tasks[i].name; document.getElementById('tStatus').value = tasks[i].status; document.getElementById('modal').style.display='flex'; }
        function deleteTask(i) { if(confirm("Delete?")) { tasks.splice(i, 1); render(); } }

        window.onload = function() {
            new Chart(document.getElementById('dashChart'), { type:'line', data:{labels:['W1','W2','W3','W4'], datasets:[{label:'Progress', data:[65,72,85,92], borderColor:'#3b82f6', tension:0.3}]}});
            new Chart(document.getElementById('skillChart'), { type: 'doughnut', data: { labels: ['Comm', 'Lead', 'Tech'], datasets: [{ data: [85, 78, 92], backgroundColor: ['#3b82f6', '#22c55e', '#a855f7'], cutout: '70%' }] } });

            new Chart(document.getElementById('radarChart'), {
                type: 'radar',
                data: {
                    labels: ['Comm', 'Problem', 'Lead', 'Crit', 'Tech', 'Time'],
                    datasets: [{ data: [85, 92, 78, 88, 82, 75], backgroundColor: 'rgba(0, 168, 132, 0.2)', borderColor: '#00a884', borderWidth: 2 }]
                },
                options: { plugins: { legend: { display: false } }, scales: { r: { suggestedMin: 0, suggestedMax: 100 } } }
            });

          new Chart(document.getElementById('weekChart'), { type:'line', data:{labels:['M','T','W','T','F'], datasets:[{label:'Attendance', data:[100,95,100,85,92], borderColor:'#22c55e'}]}});
            new Chart(document.getElementById('monthChart'), { type:'bar', data:{labels:['Jan','Feb','Mar'], datasets:[{label:'%', data:[82,88,92], backgroundColor:'#3b82f6'}]}});

            const cal = document.getElementById('calGrid');
            for(let i=1; i<=31; i++) {
                let color = (i===4 || i===18) ? "var(--red)" : (i===6 || i===25) ? "var(--orange)" : "var(--green)";
                cal.innerHTML += `<div class="day-box" style="background:${color}">${i}</div>`;
            }
            render();
        };
    </script>
